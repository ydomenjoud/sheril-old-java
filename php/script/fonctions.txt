function base_flotte($base, $commandant)
{
    $tab = [];
    $result = mysql_query("SELECT * FROM diviser_flotte WHERE NUMERO='$commandant'");
    $nb_lignes = mysql_num_rows($result);
    for ($i = 0; $i < $nb_lignes; $i++) {

        $num = $rf[1] + 1;
        $tab[10000 + $rf[3]] = "$rf[2] (divisée à partir de la flotte $num)";
    }
    return $tab;
}

function r_base2($result, $mod)
{
    $nb_lignes = mysql_num_rows($result);
    if(!$nb_lignes) {
        echo mysql_error();
    }
    if ($nb_lignes == 0) return "";
    $rf = mysql_fetch_row($result);
    return explode(",", trim(ereg_replace("#", "'", $rf[1 - $mod])));
}

function r_base($result, $mod)
{
    $nb_lignes = mysql_num_rows($result);
    if ($nb_lignes == 0) return "";
    $rf = mysql_fetch_row($result);
    $tra = explode(",", trim(ereg_replace("#", "'", $rf[1 - $mod])));
    $tra2 = explode(",", trim(ereg_replace("#", "'", $rf[2 - $mod])));
    for ($i = 0; $i < sizeof($tra); $i++)
        $t[$tra[$i]] = $tra2[$i];
    return $t;
}

function fusion_tab($tab1, $tab2)
{
    $j = sizeof($tab1);
    for ($i = 0; $i < sizeof($tab2); $i++)
        $tab1[$j + $i] = $tab2[$i];
    return $tab1;
}

function fusion_tab2($tab1, $tab2)
{
    $j = sizeof($tab1);
    $k = array_keys($tab2);
    for ($i = 0; $i < sizeof($k); $i++) {
        $n = $k[$i];
        $tab1[$n] = $tab2[$n];
    }
    return $tab1;
}

function select1($nv, $t, $class='select', $multiple=false, $maxWidth = '200')
{

    echo("<SELECT SIZE=\"".($multiple ? 10 : 1)."\" NAME=\"$nv".($multiple ? '[]' : '')."\" style='max-width: {$maxWidth}px; display: inline-flex;' class='$class' ".($multiple ? 'multiple' : ''). ">");
    $k = array_keys($t);
    // Récupération de la valeur postée une seule fois pour plus de clarté
    $postedValue = isset($_POST[$nv]) ? $_POST[$nv] : null;
    for ($i = 0; $i < sizeof($k); $i++) {
        $j = $k[$i];
        $selected = '';

        if ($multiple) {
            // Cas multiple : on vérifie si la clé appartient au tableau
            if (is_array($postedValue) && in_array($j, $postedValue)) $selected = 'selected ';
        } else {
            // Cas simple : comparaison classique
            if ($postedValue == $j) $selected = 'selected ';
        }

        echo("<OPTION ".$selected." ");
        echo("VALUE=\"$j\">$t[$j]");
    }
    echo("</SELECT>");
}

function baseList($table, $id=null, $where=null)
{
    $query = "SELECT * FROM $table";
    if($where) {
        $query .= " WHERE $where";
    }
    $resultList = mysql_query($query);
    if (!$resultList) {
        echo mysql_error();
    }
    $list = [];
    while ($row = mysql_fetch_assoc($resultList)) {
        if($id){
            $list[$row[$id]] = $row;
        }
        else $list[] = $row;
    }
    return $list;
}

/**
 * Retourne l'index de la colonne "id" dans une table donnée.
 * * @param string $nom_table Le nom de la table à vérifier
 * @param resource $connexion (Optionnel) La ressource de connexion MySQL
 * @return int L'index (0-based) ou -1 si non trouvé
 */
function get_id_column_index($nom_table) {
    // 1. Exécuter une requête légère pour obtenir la structure (LIMIT 0 suffit)
    $sql = "SELECT * FROM " . mysql_real_escape_string($nom_table) . " LIMIT 0";
    $result = mysql_query($sql);

    if (!$result) {
        return -1; // La table n'existe probablement pas ou erreur SQL
    }

    // 2. Récupérer le nombre total de colonnes
    $nb_champs = mysql_num_fields($result);

    // 3. Parcourir les colonnes pour trouver celle nommée "id"
    for ($i = 0; $i < $nb_champs; $i++) {
        $nom_colonne = mysql_field_name($result, $i);

        // Comparaison insensible à la casse (optionnel, selon vos besoins)
        if (strtolower($nom_colonne) === 'id') {
            return $i; // On a trouvé l'index !
        }
    }

    // 4. Si on sort de la boucle sans avoir trouvé
    return -1;
}

function getCommandantHTML($num) {
    if (empty($num)) return "Inconnu";
    global $base;
    $num = (int)$num;
    $sql = "SELECT NOM, NUMERO, RACE FROM aa_registre WHERE NUMERO = $num";
    $res = @mysql($base, $sql);
    if ($res && $row = @mysql_fetch_assoc($res)) {
        return "<span class='race{$row['RACE']}'>{$row['NOM']} ({$row['NUMERO']})</span>";
    }
    return "Inconnu (" . $num . ")";
}
