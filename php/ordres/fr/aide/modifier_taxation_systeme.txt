function calculerChanceRevolteGlobale($stabilite,$nbPlanetes, $arrondi = 2): float
{
    // 1. Convertir la stabilité en valeur décimale (ex: 95 devient 0.95)
    $stabiliteDecimale = $stabilite / 100;

    // 2. Calculer la probabilité qu'AUCUNE planète ne se révolte
    $chanceAucuneRevolte = pow($stabiliteDecimale, $nbPlanetes);

    // 3. En déduire la probabilité qu'au moins une se révolte
    $chanceAuMoinsUneRevolte = 1 - $chanceAucuneRevolte;

    // 4. Retourner le résultat sous forme de pourcentage arrondi
    return round($chanceAuMoinsUneRevolte * 100, $arrondi);
}
